# Set msg to point at the current top of the heap.
# Every task has a stack and a heap, which grow towards each other.
# All variables are 16-bit integers, which can also be interpreted as
# as pointers.
heap =msg

loop {
    # Call getc, putting this task to sleep & waking it up once a
    # byte is available on its stdin
    getc =c

    # Push c onto the heap as a single byte
    c hpushc

    # If c was a NUL byte, exit the loop
    c while
}

# At this point, msg points to a C-style string on the heap!..
# Now we're going to repeatedly echo it back over stdout.
# For fun, we'll keep track of how many times we echo it.
0 =n_echoes

loop {

    # Set ptr to msg, i.e. the start of the string
    msg =ptr

    # Loop over the characters in the string
    loop {
        # Dereference ptr, getting a byte, and store that in c
        ptr $c =c

        # Break on NUL byte
        c while

        # Call putc, putting c in this task's stdout buffer, then
        # putting this task to sleep & waking it up once the byte has
        # been consumed
        c putc

        # Increment ptr
        ptr 1+ =ptr
    }

    # Increment our echo counter
    n_echoes 1+ =n_echoes

    # Logging is sent directly through the USB to the Python client,
    # using low-level assembly routines.
    # That is, tasks don't have a separate stderr stream, rather logging
    # is atomic and bypasses the OS.
    # Why?.. because logging is for debugging purposes, of course!
    # It would be harder to debug the OS if logging was tied up with it...
    # Anyway, if we want to log a dynamic message, we don't build a string
    # on the heap!.. we log it in parts.
    # We can use an atomic block to make sure other tasks don't log (or do
    # anything else) while we're logging.
    atomic {
        "Echoed " logs
        msg logs
        ' ' logc
        n_echoes logi
        " times!\n" logs
    }
}
