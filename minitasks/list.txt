
# The end-of-list marker
def nil { 0 }

# Create a list from a head (i.e. data element) and tail (i.e. pointer
# to the next list node, or nil if this is the list's final element).
def list.create head tail : _ {
    # Push a pointer to current top of heap onto the stack
    heap

    # Push head and tail onto the head
    head hpush
    tail hpush

    # The pointer to the old top of heap is still on top of the stack,
    # and so is returned!..
}

# NOTE: no memory is actually freed in this particular list implementation.
# But we could implement it if we wanted to, by somehow marking list nodes
# as being in use or freed, possibly shrinking the heap using "hdrop"
# and/or "hpop", etc.
def list.free _ : { drop }

# Functions to get/set the head and tail of a list.
# For speed & brevity, the stack is being manipulated directly, instead
# of using local variables.
def list.head _ : _ { $ }
def list.tail _ : _ { 2+ $ }
def list.set_head _ _ : { =$ }
def list.set_tail _ _ : { swap 2+ swap =$ }

# Given a list, jump to its "n"th element.
def list.at list n : _ {
    loop {
        n 0 > while
        list list.tail =list
        n 1- =n
    }
    list
}

# Get and set the "n"th element of a list.
# Note that not all function inputs are explicity listed; e.g. list.get
# expects 2 inputs on top of the stack (to be used by list.at), but since
# it doesn't need to store them in local variables, it doesn't bother
# mentioning them.
def list.get _ _ : _ { list.at list.get_head }
def list.set _ _ x : { list.at x list.set_head }

# Build the list [10, 20, 30]
10 20 30 nil list.create list.create list.create =x

# Get some list elements
x 0 list.get 10 == assert
x 1 list.get 20 == assert
x 2 list.get 30 == assert

# Modify a list element (the list now looks like: [10, 99, 30])
x 1 99 list.set
x 1 list.get 99 == assert
